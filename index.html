Khoảng cách thời gian trong timepicker

Cách nhau 15 phút


const dtp14 = new GDatetimepicker('#dtp14', {
	format: 'DD/MM/YYYY HH:mm',
	step: 15,
});

Thay đổi khoảng lựa chọn thời gian theo ngày

Thứ 2-6: 16:00 - 21:00

Thứ 7, CN: 08:00 - 21:00


const dtp15 = new GDatetimepicker('#dtp15', {
	format: 'DD/MM/YYYY HH:mm',
	step: 30,
	beforeShowTime: function (currentDateTime) {
		let day = currentDateTime.getDay();
		if (day == 6 || day == 0) {
			return {
				minTime: '8:00',
				maxTime: '21:00',
				disabledTimes: ['12:00', '12:30', '13:00', '13:30']
			};
		} else {
			return {
				minTime: '16:00',
				maxTime: '21:00',
			};
		}
	},
});
